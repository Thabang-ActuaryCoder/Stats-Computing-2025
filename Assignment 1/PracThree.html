# 2. Calculate average delays per airline on these routes

multi_airline_routes <- flights |>
  group_by(origin, dest) |>
  summarise(n_airlines = n_distinct(carrier)) |>
  filter(n_airlines > 1) |>
  ungroup()

route_delays <- flights |>
  semi_join(multi_airline_routes, by = c("origin", "dest")) |>
  group_by(origin, dest, carrier) |>
  summarise(avg_arr_delay = mean(arr_delay, na.rm = TRUE), .groups = "drop") |>
  left_join(airlines, by = "carrier") |>
  select(origin, dest, name, avg_arr_delay)

route_delays |>
  arrange(origin, dest) |>
  head() |>

# 3. Identify best/worst performing airlines per route

route_comparison <- route_delays |>
  group_by(origin, dest) |>
  summarise(
    best_airline = name[which.min(avg_arr_delay)],
    best_delay = min(avg_arr_delay),
    worst_airline = name[which.max(avg_arr_delay)],
    worst_delay = max(avg_arr_delay),
    delay_diff = worst_delay - best_delay,
    .groups = "drop"
  )

  arrange(desc(delay_diff)) |>
